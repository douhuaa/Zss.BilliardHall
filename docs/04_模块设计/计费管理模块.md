    public BillCalculationResult Calculate(
        TimeSpan duration,
        decimal hourlyRate,
        List<PricingRule> rules)
    {
        var hours = Math.Ceiling(duration.TotalHours);
        var baseAmount = (decimal)hours * hourlyRate;
        
        var items = new List<BillItem>
        {
            new BillItem
            {
                Description = $"æ¡Œå°è´¹ç”¨ï¼ˆ{hours:F2}å°æ—¶ Ã— {hourlyRate:F2}å…ƒ/å°æ—¶ï¼‰",

        // åº”ç”¨å³°æ—¶ä»·æ ¼ç­–ç•¥
        var now = DateTime.Now.TimeOfDay;

            r.PeakTimeStart.HasValue && r.PeakTimeEnd.HasValue &&
            now >= r.PeakTimeStart.Value && now <= r.PeakTimeEnd.Value);
        
        if (peakTimeRule != null && peakTimeRule.PeakTimeMultiplier > 1)
        {
            var surcharge = baseAmount * (peakTimeRule.PeakTimeMultiplier - 1);

            {
                Description = $"é«˜å³°æ—¶æ®µåŠ ä»·ï¼ˆ{peakTimeRule.PeakTimeMultiplier:P0}ï¼‰",
                Amount = surcharge,
                Type = BillItemType.TableFee

        }
        
        return new BillCalculationResult
        {
            BaseAmount = baseAmount,
            Items = items
        };
    }
}
